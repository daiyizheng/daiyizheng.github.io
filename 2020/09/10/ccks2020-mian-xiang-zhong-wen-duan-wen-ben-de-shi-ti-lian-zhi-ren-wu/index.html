CCKS2020面向中文短文本的实体链指任务

实体歧义消解
比赛
发布日期:   2020-09-10

实体链指

任务介绍

面向中文短文本的实体链指，简称 EL（Entity Linking）。即对于给定的一个中文短文本（如搜索 Query、微博、对话内容、文章/视频/图片的标题等），EL 将其中的实体与给定知识库中对应的实体进行关联。针对中文短文本的实体链指存在很大的挑战，主要原因如下：

（1）口语化严重，导致实体歧义消解困难；
（2）短文本上下文语境不丰富，须对上下文语境进行精准理解；
（3）相比英文，中文由于语言自身的特点，在短文本的链指问题上更有挑战。

任务详情

此次任务的输入输出定义如下：

输入：中文短文本以及该短文本中的实体集合。

输出：输出文本此中文短文本的实体链指结果。每个结果包含：实体 mention、在中文短文本中的位置偏移、其在给定知识库中的 id，如果为 NIL 情况，需要再给出实体的上位概念类型

示例输入：
&#123; "text_id":"1", "text":"《琅琊榜》海宴_【原创小说|权谋小说】", "mention_data":[ &#123;  "mention":"琅琊榜",  "offset":"1" &#125;,   &#123;"mention":"海宴",  "offset":"5"    &#125;,   &#123; "mention":"原创小说",   "offset":"9" &#125;,&#123; "mention":"权谋小说",   "offset":"14"  &#125; ]&#125;
示例输出：
&#123; "text_id":"1",    "text":"《琅琊榜》海宴_【原创小说|权谋小说】",  "mention_data":[ &#123; "kb_id":"2135131",            "mention":"琅琊榜",   "offset":"1"   &#125;,&#123; "kb_id":"10572965",   "mention":"海宴",     "offset":"5"  &#125;,     &#123;  "kb_id":"215143",   "mention":"原创小说",  "offset":"9"  &#125;, &#123; "kb_id":" NIL_Work ",  "mention":"权谋小说",   "offset":"14" &#125; ]&#125;

数据介绍

知识库

&#123;"alias": [], "subject_id": "10001", "data": [&#123;"predicate": "政治面貌", "object": "中共党员"&#125;, &#123;"predicate": "义项描述", "object": "潜山县塔畈乡副主任科员、纪委副书记"&#125;, &#123;"predicate": "性别", "object": "男"&#125;, &#123;"predicate": "学历", "object": "大专"&#125;, &#123;"predicate": "中文名", "object": "张健"&#125;], "type": "Person", "subject": "张健"&#125;

知识库中的每个实体都包含alias, 对应subject可能的别名一个subject_id(知识库id)，一个subject名称，实体的别名(主语)，对应的概念类型，以及与此实体相关的一系列二元组< predicate，object>（<属性，属性值>）或者叫（谓语，宾语）信息形式。知识库中每行代表知识库的一条记录（一个实体信息），每条记录为json数据格式。

数据集

标注数据集由训练集、验证集和测试集组成，整体标注数据大约10万条左右。
标注数据集中每条数据的格式为：

&#123;"text_id": "1", "text": "天下没有不散的宴席 - ╰つ雲中帆╰つ", "mention_data": [&#123;"kb_id": "28270", "mention": "天下没有不散的宴席", "offset": "0"&#125;, &#123;"kb_id": "NIL_Other", "mention": "╰つ雲中帆╰つ", "offset": "12"&#125;]&#125;

kb_id对应知识库subject_id，mention对应的是实体．

logger = logging.getLogger(__name__)
PICKLE_PATH = "middle_data/kb/"
entity_to_kbids = defaultdict(set) ## 实体名称对应KDID表
kbid_to_entities = dict()## KBID对应的实体名称列表
kbid_to_text = dict() ## KDID对应属性文本
kbid_to_types = dict()## KDID对应实体类型表
kbid_to_predicates = dict() # KBID对应的属性列表
idx_to_type = list() ## 索引类型映射列表
type_to_idx = dict() ##类型索引映射字典
with open("/media/daiyizheng/SSD/data/CCKS/2020/ccks2020面向中文短文本的实体链指任务/kb.json", 'r', encoding="utf-8") as f:
    for line in tqdm(f.readlines()):
        line = json.loads(line)
        entities = set(line['alias'])
        kbid = line['subject_id']
        type = line['type']
        subject = line['subject']
        entities.add(subject)
        for entity in entities:
            entity_to_kbids[entity].add(kbid) #实体名称对应KDID表
        kbid_to_entities[kbid] = entities

        text_list, predicate_list = [], []

        for x in line['data']:
            # data中的predicate和object拼接
            text_list.append(":".join([x['predicate'].strip(), x['object'].strip()]))
            predicate_list.append(x['predicate'].strip())

        random.shuffle(text_list) # 对属性文本随机打乱顺序

        kbid_to_text[kbid] = ' '.join(text_list)
        kbid_to_predicates[kbid] = predicate_list

        for c in ['\r', '\t', '\n']:
            kbid_to_text[kbid] = kbid_to_text[kbid].replace(c, "")

        type_list = line['type'].split('|')
        kbid_to_types[kbid] = type_list

        for t in type_list:
            if t not in type_to_idx:
                type_to_idx[t] = len(type_to_idx)
                idx_to_type.append(t)

    # 保存pickle文件
    pd.to_pickle(entity_to_kbids, PICKLE_PATH + 'ENTITY_TO_KBIDS.pkl')
    pd.to_pickle(kbid_to_entities, PICKLE_PATH + 'KBID_TO_ENTITIES.pkl')
    pd.to_pickle(kbid_to_text, PICKLE_PATH + 'KBID_TO_TEXT.pkl')
    pd.to_pickle(kbid_to_types, PICKLE_PATH + 'KBID_TO_TYPES.pkl')
    pd.to_pickle(kbid_to_predicates, PICKLE_PATH + 'KBID_TO_PREDICATES.pkl')
    pd.to_pickle(idx_to_type, PICKLE_PATH + 'IDX_TO_TYPE.pkl')
    pd.to_pickle(type_to_idx, PICKLE_PATH + 'TYPE_TO_IDX.pkl')
    logger.info('Process Pickle File Finish.')

实体链指预处理

def generate_el_dataset(file_name, max_negs=-1):
    link_dict = defaultdict(list)
    path = os.path.join(data_dir, file_name)
    with open(path, 'r', encoding='utf-8') as f:
        for line in tqdm(f.readlines()):
            line = json.loads(line)
            for data in line['mention_data']:
                ## 对测试集特殊处理
                if 'kb_id' not in data:
                    data['kb_id'] = '0'

                # KB中不存在的实体不进行链接
                if not data['kb_id'].isdigit():
                    continue

                entity = data['mention']
                kbids = list(entity_to_kbids[entity])
                random.shuffle(kbids)

                num_negs = 0
                for kbid in kbids:
                    if num_negs >= max_negs > 0 and kbid != data['kb_id']:
                        continue
                    link_dict['text_id'].append(line['text_id'])
                    link_dict['entity'].append(entity)
                    link_dict['offset'].append(data['offset'])
                    link_dict['short'].append(line['text'])
                    link_dict['kb_id'].append(kbid)
                    link_dict['kb_text'].append(kbid_to_text[kbid])
                    link_dict['kb_predicate_num'].append(len(kbid_to_predicates[kbid]))
                    if kbid != data['kb_id']:
                        link_dict['predict'].append(0)
                        num_negs += 1
                    else:
                        link_dict['predict'].append(1)
    link_data = pd.DataFrame(link_dict)
    type_name = file_name.split('.')[0]
    output_path = os.path.join(out_dir, "el_" + type_name + ".tsv")
    link_data.to_csv(output_path, index=False, sep='\t', encoding='utf_8_sig')

负例选取

通过该实体查ENTITY_TO_KBIDS所有与kid，当数据集某一条记录kid相同时正例，不相同则是负例(同一实体，不同语义)负例最多取两条．

def generate_et_dataset(file_name):
    type_dict = defaultdict(list)
    path = os.path.join(data_dir, file_name)
    with open(path, 'r', encoding='utf-8') as f:
        for line in tqdm(f.readlines()):
            line = json.loads(line)
            for data in line['mention_data']:
                entity = data['mention']
                # 测试集特殊处理
                if 'kb_id' not in data:
                    entity_type = ['Other']
                elif data['kb_id'].isdigit():
                    entity_type = kbid_to_types[data['kb_id']]
                else:
                    entity_type = data['kb_id'].split("|")
                    for x in range(len(entity_type)):
                        entity_type[x] = entity_type[x][4:]

                for e in entity_type:
                    type_dict['text_id'].append(line['text_id'])
                    type_dict['entity'].append(entity)
                    type_dict['offset'].append(data['offset'])
                    type_dict['short_text'].append(line['text'])
                    type_dict['type'].append(e)

    type_data = pd.DataFrame(type_dict)
    output_path = os.path.join(out_dir, "et_"+file_name.split('.')[0] + ".tsv")
    type_data.to_csv(output_path, index=False, sep="\t", encoding="utf-8")

class EntityLinkingModel(nn.Module):

    def __init__(self, args):
        super().__init__()
        self.args = args
        bert_config = BertConfig.from_pretrained(args.model_name_or_path)
        bert_config.num_labels = 1
        config = &#123;"config": bert_config&#125;
        self.bert = BertForSequenceClassification.from_pretrained(args.model_name_or_path, **config) ## bert模型
        self.criterion = nn.BCEWithLogitsLoss

    def forward(self, input_ids, attention_mask, token_type_ids):
        output = self.bert(input_ids=input_ids, attention_mask=attention_mask, token_type_ids=token_type_ids)
        return output
